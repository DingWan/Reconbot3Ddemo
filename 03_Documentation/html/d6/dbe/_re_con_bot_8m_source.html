43  %% The buildTrajectory Function
44  % The _buildTrajectory_ method builds the _FollowJointTrajectoryGoal_
45  % message which is the message it must be sent to the ReConBot
46  % controller. The function returns a trajectory of the type
47  % _FollowJointTrajectoryGoal_, this trajectory must be passed to
48  % the object property _tj_. Before creating the trajectory, the
49  % trajectory points vector must be specified. This vector hast to be
50  % passed to the instance (_objName_) throught the propery _points_.
51  % For instance, for two desired trajectory points:
52  %
53  % >> points = [ 3 4 2 3 7 9 0 9 9 4; 7 2 3 4 2 4 3 4 2 8];
54  %
55  % and then,
56  %
57  % >> objName.points = points
58  %
59  % The joint names have to be defined into the object property
60  % _jointNames_:
61  %
62  % >> objName.jointNames = {'joint_1', 'joint_2', 'joint_3'}
63  %
64  % Finally, the trajectory message can be built:
65  %
66  % >> objName.tj = buildTrajectory(objName)
67  %
68  function traj=buildTrajectory(obj)
69  traj = rosmessage('control_msgs/FollowJointTrajectoryGoal');
70  traj.Trajectory.JointNames = obj.jointNames;
71  [m,n]=size(obj.points);
72  for i=1:m
73  positions = obj.points(i,1:3:n-1);
74  velocities = obj.points(i,2:3:n);
75  accelerations = obj.points(i,3:3:n);
76  timefromstart = rosduration(obj.points(i,n));
77  points_msg = rosmessage('trajectory_msgs/JointTrajectoryPoint');
78  points_msg.Positions = positions;
79  points_msg.Velocities = velocities;
80  points_msg.Accelerations = accelerations;
81  points_msg.TimeFromStart = timefromstart;
82  traj.Trajectory.Points(i) = points_msg;
83  clear points_msg;
84  if i==m
85  disp('FollowJointTrajectoyGoal message successfully built.')
86  end
87  end
88  end
