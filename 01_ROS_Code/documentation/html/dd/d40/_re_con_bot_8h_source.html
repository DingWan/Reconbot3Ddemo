190 control_msgs::FollowJointTrajectoryGoal ReConBotPub::buildTrajectory(){
191  ROS_INFO("Start building the Trajectory Vector...");
192  for (size_t i = 0; i < nMotors; i++) {
193  std::stringstream ss;
194  ss << motorsActive[i];
195  goal.trajectory.joint_names.push_back("joint_"+ ss.str());
196  }
197  //goal.trajectory.joint_names.push_back("joint_1");
198  //goal.trajectory.joint_names.push_back("joint_2");
199  //goal.trajectory.joint_names.push_back("joint_3");
200 
201  std::fstream pathfile(sourceFile.c_str(), std::ios_base::in);
202  int number_of_lines = 0;
203  std::string line;
204  std::ifstream myfile(sourceFile.c_str());
205 
206  while (std::getline(myfile, line)){
207  ++number_of_lines;
208  }
209  goal.trajectory.points.resize(number_of_lines);
210  cont =0;
211  var =0;
212  while (pathfile >> pos && ros::ok()) {
213  trajectoryPoints.resize(number_of_lines*(nMotors*3+1));
214  trajectoryPoints[var]=pos;
215  ++var;
216  if (var == nMotors*3 + 1) {
217  j = 0;
218  for (size_t i = 0; i < nMotors*3; i=i+3) {
219  goal.trajectory.points[cont].positions.resize(nMotors);
220  goal.trajectory.points[cont].positions[j] = trajectoryPoints[i];
221  ++j;
222  }
223  j = 0;
224  for (size_t i = 1; i < nMotors*3; i=i+3) {
225  goal.trajectory.points[cont].velocities.resize(nMotors);
226  goal.trajectory.points[cont].velocities[j] = trajectoryPoints[i];
227  ++j;/* condition */
228  }
229  j = 0;
230  for (size_t i = 2; i < nMotors*3; i=i+3) {
231  goal.trajectory.points[cont].accelerations.resize(nMotors);
232  goal.trajectory.points[cont].accelerations[j] = trajectoryPoints[i];
233  if (i == nMotors*3 - 1) {
234  goal.trajectory.points[cont].time_from_start = ros::Duration(trajectoryPoints[i+1]);
235  }
236  ++j;
237  }
238  var = 0;
239  ++cont;
240  ROS_INFO("New trajectory point created...");
241  }
242  }
243  ROS_INFO("Trajectory vector was successfully built....");
244  return goal;
245 }
